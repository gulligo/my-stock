async function autoFetchPrice() { const name = document.getElementById('stockName').value.trim(); const priceInput = document.getElementById('currentPrice'); if (!name) return; priceInput.value = "조회 중..."; try { const response = await fetch(`https://query1.finance.yahoo.com/v1/finance/search?q=${encodeURIComponent(name)}&lang=ko-KR`); const data = await response.json(); if (data.quotes && data.quotes.length > 0) { const symbol = data.quotes[0].symbol; const priceResponse = await fetch(`https://query1.finance.yahoo.com/v8/finance/chart/${symbol}?interval=1m&range=1d`); const priceData = await priceResponse.json(); const nowPrice = priceData.chart.result[0].meta.regularMarketPrice; // 이제 진짜 실시간 가격만 표시합니다! priceInput.value = Math.floor(nowPrice).toLocaleString() + " 원"; } else { priceInput.value = "종목 확인 불가"; } } catch (error) { console.error(error); // 실패했을 때 엉뚱한 숫자가 나오지 않도록 "재시도"라고만 띄웁니다. priceInput.value = "다시 입력해주세요"; } }
